/**
 * Fetch Mocked Data
 *
 * @param { number } userId
 * @return { data }
 */
export async function getUserData(userId) {
    try {
        const response = await fetch(`${process.env.PUBLIC_URL}/user/${userId}/user.json`)
        const {data} = await response.json()
        return data
    } catch(err) {
        return err
    }
}
